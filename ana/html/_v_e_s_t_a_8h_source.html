202  int upolindex = (vstr_.size() -1)/2 ;
203  vector <double> vni, vpi, vui, vq;
204  TGraph *g = new TGraph();
205  for (int i = 0; i < vgaus_[npolindex]->GetN(); i++){
206  double q, ni, pi, ui;
207  vgaus_[npolindex] -> GetPoint(i, q, ni);
208  vgaus_[ppolindex] -> GetPoint(i, q, pi);
209  vgaus_[upolindex] -> GetPoint(i, q, ui);
210  double ana = 0.0;
211  double k = 3.89;
212  switch (mode){
213  case 0: //Asymmetry
214  ni += 1000;
215  pi += 1000;
216  ana = (ni - pi)/(ni + pi);
217  break;
218  case 1: //mix
219  ana = (ni - pi)/2.0/pol/2.0/k;
220  break;
221  case 2: //self
222  ana = (ni - pi - 2*ui)/2.0/pow(pol, 2)/pow(k,2);
223  break;
224  case 3: //cross
225  ana = ((ni - pi)/2.0/pol/2.0/k) - ((ni - pi - 2*ui)/2.0/pow(pol, 2)/pow(k,2));
226  }
227  g->SetPoint(g->GetN(), q, ana);
228  }
229  return g;
230 }
